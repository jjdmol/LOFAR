// This file contains the reserved unique protocol id's specified for MAC controller with
// at least one GCF (M)SPP port. If necessary add your new unique protocol id in the table 
// below. If you want change or delete id's, please contact the MAC integrator first. 
//
// Note: The enum below can be found in the GCF_Protocols.h
// Note: The id "F_APL_PROTOCOL" (value 10) will commonly be used for test applications 
//       in GCF. It is recommended to NOT use this 'id' in APL.
// Note: The id's are specified in <ServerName>_Protocol.prot files. 

// $Id$

enum 
{ 
  F_FSM_PROTOCOL = 1,   // state machine protocol (encoded in msb)
  F_PORT_PROTOCOL,      // port connection protocol
  F_GCF_PROTOCOL,       // GCF specific protocol numbers start here
  F_APL_PROTOCOL = 10,  // APPlication specific protocol numbers start here
};
