include(LofarCTest)

include(MACAddProtocol)

mac_add_protocol(Echo_Protocol "${PACKAGE_SOURCE_DIR}/autogen")

lofar_add_test(tProtocol tProtocol.cc Echo_Protocol.cc testprotocol.cc)
lofar_add_test(tClone tClone.cc Clone_Protocol.cc)
